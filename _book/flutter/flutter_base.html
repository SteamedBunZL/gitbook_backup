Dart null 检查最佳实践

跟kotlin一样 使用?.

?.

kotlin中的?: 在dart中是??

bool isConnected(a,b){
    bool outConn = outgoing[a]?.contains(b) ?? false;//如果??前面为null 就执行赋值??后面的
}


  print(null ?? false);//false null == null 所以赋值 false
  print(false ?? 11);//false  false不为null 所以不执行11 执行前面的
  print(true ?? false);//true
